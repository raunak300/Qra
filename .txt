const mongoose = require('mongoose');
const db = process.env.DB_URL;
const bcrypt = require('bcrypt');


mongoose.connect(db).
    then(() => {
        console.log("connected to db")
    }).catch((err) => {
        console.log("MONGODB connection error:", err);
    })
const userSchema = new mongoose.Schema({
    fullName: {
        type: String,
        required: true,
    },
    email: {
        type: String,
        required: true,
        unique: true
    },
    password: {
        type: String,
        required: true,
    },
    userName: {
        type: String,
        required: false,
        unique:true
    },
    userImg: {
        type: String,
        default:null
    },
    uploadedImages: {
        type: [String],
        default: [],
    },
    createdAt: {
        type: Date,
        default: Date.now
    },
})

userSchema.pre('save', async function(next) {
    try {
        // Only hash the password if it has been modified (or is new)
        if (!this.isModified('password')) {
            return next();
        }
        const salt = await bcrypt.genSalt(10);
        const hashedPassword = await bcrypt.hash(this.password, salt);
        this.password = hashedPassword;
        next();
    } catch (err) {
        console.log(`This is an error on hashing password: ${err}`);
        next(err); // Pass the error to the next middleware
    }
});
const User=mongoose.model('User',userSchema);


module.exports= User;

const User = require('../Model/userModel');
var jwt = require('jsonwebtoken');
const bcrypt = require('bcrypt');

const signUp = async (req, res) => {

    try {
        const { fullName, email, password } = req.body;
        console.log("this is data received on backend at /signup check userController:", req.body);
        const newUser = await new User({
            fullName,
            email,
            password
        })
        await newUser.save();
        try {
            const token = jwt.sign({ id: newUser._id, email: newUser.email }, process.env.JWT_SECRET, { expiresIn: '1h' });
            res.cookie('token', token, {
                secure: true, // Use secure cookies in production
                maxAge: 60 * 60 * 1000,
                sameSite: "None"
            });
            console.log("token created");

        } catch (error) {
            console.log(`error while creating token/cookie in signup: ${error}`)
        }
        res.status(201).send({ message: "User SignUp Succesful", user: newUser })
    } catch (error) {
        console.log("error in SignUp", error);
        res.status(500).json({ message: 'Internal server error in signup' });
    }

}

const login = async (req, res) => {
    try {
        const { email, password } = req.body;
        const user =await  User.findOne({ email: email });
        if (user) {
            console.log("user email found yay from login");
            const check =await  bcrypt.compare(password, user.password);
            if (!check) {
                return res.status(401).send({ message: "user does not exist invalid credentials at login" });
            }
            else {
                try {
                    
                    const token = jwt.sign({ id: user._id, email: user.email }, process.env.JWT_SECRET, { expiresIn: '1h' });

                  
                    res.cookie('token', token, {
                        secure: true, 
                        maxAge: 60 * 60 * 1000, 
                        sameSite: "None"
                    });
                    console.log("token created and cookie set as well");
                    return res.status(200).send({message:"login succesfule",data:user});

                } catch (error) {
                    console.log(`error while creating token/cookie in login: ${error}`);
                    return res.status(500).send({ message: 'Login successful, but failed to create session. Please try again.' });
                }
            }
        } else {
            return res.status(401).send({ message: "user does not exist invalid credentials at login" });
        }
    } catch (error) {
        console.log(`there is error in here ${error}`)
        return res.status(500).send("error in login")
    }
}


module.exports = { signUp, login };
